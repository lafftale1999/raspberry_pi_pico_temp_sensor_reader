31int bm280_read_data(const bm280_handle_t handle)
32{
33
34 uint8_t buf[BM280_READ_VALUES_REG_LEN];
35 uint8_t register_start = BM280_READ_VALUES_REG_START;
36
37 if(i2c_read_data(&handle->i2c_address, &register_start, buf, BM280_READ_VALUES_REG_LEN) != 0) {
38 BM280_LOG("Unable to read data from registers.\n");
39 return 1;
40 }
41 handle->press_raw = (buf[0] << 12) | (buf[1] << 4) | (buf[2] >> 4);
42 handle->temp_raw = (buf[3] << 12) | (buf[4] << 4) | (buf[5] >> 4);
43 handle->hum_raw = (buf[6] << 8) | buf[7];
44
45 if(bm280_convert_measurements(handle) != PICO_W_OK) {
46 BM280_LOG("Unable to convert values.\n");
47 return 1;
48 }
49
50 bm280_parse_json(handle);
51
52 BM280_LOG("Temp: %0.1fC | Humidity: %0.1f%% | Pressure: %0.1fhPa\n", (float)handle->temperature / 100.0f, (float)handle->humidity / 1024.0f, ((float)handle->pressure / 256.0f) / 100.0f);
53
54 return 0;
55}

68int bm280_init(bm280_handle_t *handle, const uint8_t device_address, const BM280_READING_INTERVALS_MS interval)
69{
70 bm280_handle_t p_handle = malloc(sizeof(struct bm280_handle_internal));
71
72 if (p_handle == NULL) {
73 BM280_LOG("Unable to allocate handle on heap\n");
74 goto exit;
75 }
76
77 p_handle->i2c_address = device_address;
78 p_handle->address_length = BM280_ADDRESS_LENGTH;
79
80 if (bm280_calibration(p_handle) != PICO_W_OK) {
81 goto exit;
82 }
83
84 uint8_t humidity_set_up[2] = {
85 BM280_CONTROL_HUMIDITY,
86 BM280_SET_HUM_OSRS(BM280_OSRS_X1)
87 };
88
89 uint8_t measurement_set_up[3] = {
90 BM280_CONTROL_MEASURE,
91 BM280_SET_TEMP_OSRS(BM280_OSRS_X2) | BM280_SET_PRESS_OSRS(BM280_OSRS_X16) | BM280_MODE_NORMAL
92 };
93
94 uint8_t config_set_up[2] = {
95 BM280_CONFIG_REGISTER,
96 BM280_SET_STANDBY_MS(interval) | BM280_SET_IIR_FILTER(BM280_SET_FILTER_16)
97 };
98
99 // Making sure that all configurations are set up correctly
100 if (i2c_write_data(&p_handle->i2c_address, humidity_set_up, 2, true) != 0 ||
101 i2c_write_data(&p_handle->i2c_address, config_set_up, 2, true) != 0 ||
102 i2c_write_data(&p_handle->i2c_address, measurement_set_up, 2, false) != 0){
103 BM280_LOG("Unable to correctly set up configurations on bm280.\n");
104 goto exit;
105 }
106
107 *handle = p_handle;
108
109 return 0;
110
111exit:
112 BM280_LOG("Initialization of bm280 failed\n");
113
114 if (p_handle != NULL) {
115 free(p_handle);
116 }
117
118 return 1;
119}
